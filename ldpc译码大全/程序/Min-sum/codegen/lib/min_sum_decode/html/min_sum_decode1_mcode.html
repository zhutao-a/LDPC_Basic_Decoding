1%软判决译码
 2function [MM,cycle]=min_sum_decode(H,y,p)
 3
 4I=1;%迭代次数计数器
 5Imax=20;%规定最大迭代次数
 6
 7[m,n]=size(H);
 8%n=H矩阵列数/信息节点数
 9%m=H矩阵行数/校验节点数
10
11z=zeros(1,n);
12r=zeros(1,n);
13L=zeros(1,n);
14
15M=zeros(m,n);
16E=zeros(m,n);
17%r=y;
18%计算对数似然比
19for ii=1:n
20    if y(ii)==1
21        r(ii)=log(p/(1-p));
22    else
23        r(ii)=log((1-p)/p);
24    end
25end
26
27%初始化M矩阵
28for ii=1:n
29    for jj=1:m
30        M(jj,ii)=r(ii);
31    end
32end
33M=(H.*M);
34%迭代
35while 1==1
36  
37    for jj=1:m
38        position1=find(M(jj,:));
39        Mtemp=zeros(1,length(position1));
40        for flag1=1:length(position1)
41            Mtemp(flag1)=M(jj,position1(flag1));
42        end
43
44        for ii=1:n
45            if H(jj,ii)==0
46                E(jj,ii)=0;
47            else
48                E(jj,ii)=prod(sign(Mtemp))/sign(M(jj,ii))*min(abs(Mtemp));
49            end
50        end
51    end
52    
53    %更新信息节点,Test
54    for ii=1:n
55        L(ii)=r(ii)+sum(E(:,ii));
56        if L(ii)<=0
57            z(ii)=1;
58        else
59            z(ii)=0;
60        end
61    end
62
63
64    HzT=mod((H*z')',2);
65    
66    if I==Imax || isempty(find(mod(HzT,2), 1))
67        %fprintf('finish at iteration %d\n',I);
68        MM=z;
69        cycle=I;
70        return;
71    end
72    
73    I=I+1;
74    
75    for ii=1:n
76        for jj=1:m
77            if E(jj,ii) ~= 0
78                M(jj,ii)=r(ii)+sum(E(:,ii))-E(jj,ii);
79            else
80                M(jj,ii)=0;
81            end
82        end 
83    end
84    
85end
